//sorting in descending
#include<stdio.h>
int main()
{
    int i,n,j,a;
    printf("enter size of the element in array: \n");
    scanf("%d",&n);
    int array[n];
    for(i=0;i<n;i++)
    {
        scanf("%d",&array[i]);
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            if(array[i]>array[j])
            {
                a=array[i];
                array[i]=array[j];
                array[j]=a;
            }
        }
    }
    printf("After descending Sorting:\n");
    for(i=0;i<n;i++)
    {
        printf("%d\n",array[i]);
    }
    return 0;
}
