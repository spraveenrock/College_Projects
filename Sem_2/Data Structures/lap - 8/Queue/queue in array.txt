#include<stdio.h>
int front=0,last=0,queue[100];
void push()
{
    printf("Enter the element insert at beginning: ");
    scanf("%d",&queue[last]);
    last++;
}
int pop()
{
    if(front==0)
    {
        printf("Underflow\n");
    }
    else
    {
        printf("Successfully poped the element in queue\n");
        front++;
    }
    return 0;
}
int display()
{
    int i;
    for(i=front;i<last;i++)
    {
        printf(" %d\n ",queue[i]);
    }
    return 0;
}
int main()
{
    int choice;
    
    while(1)
    {
        printf("************MENU********\n1.push\n2.pop\n3.display\n4.exit\n");
        printf("Enter the option: ");
        scanf("%d",&choice);
        switch(choice)
        {
            case 1:
            {
                push();
               break;
            }
            case 2:
            {
                pop();
                break;
            }
            case 3:
            {
                display();
                break;
            }
            case 4:
            {
                return 0;
            }
        }
    
    }
    return 0;
}
