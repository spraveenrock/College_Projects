#include<stdio.h>
#include<stdlib.h>
struct node{
    int data;
    struct node* next;
    struct node* prev;
};
int main()
{
    int num,i,number,len;
    struct node* head;
    struct node* temp;
    struct node* temp2;
    struct node* newnode;
    head=(struct node*)malloc(sizeof(struct node));
    temp=malloc(sizeof(struct node));
    temp=head;
    printf("Enter the size in linked list: \n");
    scanf("%d",&num);
    temp->prev=NULL;
    newnode=malloc(sizeof(struct node));
    for(i=0;i<num;i++)
    {
        printf("element %d: ",i);
        scanf("%d",&number);
        temp->data=number;
        if(i!=num-1)
        { 
            temp->next=malloc(sizeof(struct node));
            newnode=temp;
            temp=temp->next;
            temp->prev=newnode;
        }
    }
    temp->next=NULL;
    temp=temp->prev;
    temp->next=NULL;
    temp2=temp->prev;
    temp->prev=temp2;
    printf("After deleting  at end : \n");
    while(head!=NULL)
    {
        printf("%d->",head->data);
        head=head->next;
    }
    printf("NULL\n");
    return 0;
}
